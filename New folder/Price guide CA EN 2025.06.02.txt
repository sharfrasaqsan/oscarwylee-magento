<h2 class="subtitle" style="border-left: 10px solid #66a3ff;">Tips for wearing multifocal glasses</h2>
<p class="para-content">Adjusting to a brand new pair of multifocals can be difficult. But there are some useful tips that can ease the transition and help you settle into your glasses quicker.</p>
<ul class="ordered-list">
<li>Make sure to wear your new multifocals as much as possible to encourage your eyes to adapt to your new lenses.</li>
<li>Try not to go back to your previous glasses as it can affect the progress you have made with your new multifocals.</li>
<li>Move your head when looking around, not just your eyes. By pointing your nose where you want to see you will be looking out the clearest part of the lens.</li>
<li>If you are struggling to wear them initially, try using the new glasses in a comfortable and familiar environment to help build confidence.</li>
<li>If after 1-2 weeks of using your multifocals, you are still struggling to adjust, visit an <a href="/locations/full-store-list.html">Oscar Wylee store near you</a> and our friendly team members will assist you.</li>
</ul>

          function setupCarousel() {
            const cards = document.querySelector(".cards");
            const dots = document.querySelectorAll(".dot");
            const cardWidth = cards.clientWidth / 3;
            let slideIndex = 0;
            let slideInterval;
    
            function goToSlide(index) {
              cards.style.transform = `translateX(-${index * cardWidth}px)`;
              dots.forEach((dot, idx) => {
                dot.classList.toggle("active", idx === index);
              });
            }
    
            function autoSlide() {
              slideIndex = (slideIndex + 1) % 3; // Adjust the 3 based on the number of cards
              goToSlide(slideIndex);
            }
    
            function startAutoSlide() {
              if (slideInterval) clearInterval(slideInterval);
              slideInterval = setInterval(autoSlide, 3000);
            }
    
            function stopAutoSlide() {
              if (slideInterval) clearInterval(slideInterval);
            }
    
            // Enable the carousel for responsive sizes
            function enableCarousel() {
              startAutoSlide();
    
              // Handle click on navigation dots
              dots.forEach((dot, index) => {
                dot.addEventListener("click", () => {
                  slideIndex = index;
                  goToSlide(slideIndex);
                  startAutoSlide();
                });
              });
            }
    
            // Check if the screen width is less than or equal to 768px
            if (window.matchMedia("(max-width: 768px)").matches) {
              enableCarousel();
            } else {
              // Stop sliding for desktop sizes
              stopAutoSlide();
            }
    
            // Adjust slide position on window resize
            window.addEventListener("resize", () => {
              if (window.matchMedia("(max-width: 768px)").matches) {
                enableCarousel();
              } else {
                stopAutoSlide();
                cards.style.transform = "translateX(0)"; // Reset the transform for desktop sizes
                dots.forEach((dot, index) =>
                  dot.classList.toggle("active", index === 0)
                );
                slideIndex = 0;
              }
            });
          }
    
          setupCarousel(); // Call the setupCarousel function
